سُورَةُ المُرۡسَلَاتِ
﷽
وَٱلۡمُرۡسَلَٰتِ عُرۡفٗا ۝١ فَٱلۡعَٰصِفَٰتِ عَصۡفٗا ۝٢ وَٱلنَّٰشِرَٰتِ نَشۡرٗا ۝٣ فَٱلۡفَٰرِقَٰتِ فَرۡقٗا ۝٤ فَٱلۡمُلۡقِيَٰتِ ذِكۡرًا ۝٥ عُذۡرًا أَوۡ نُذۡرًا ۝٦ إِنَّمَا تُوعَدُونَ لَوَٰقِعٞ ۝٧ فَإِذَا ٱلنُّجُومُ طُمِسَتۡ ۝٨ وَإِذَا ٱلسَّمَآءُ فُرِجَتۡ ۝٩ وَإِذَا ٱلۡجِبَالُ نُسِفَتۡ ۝١٠ وَإِذَا ٱلرُّسُلُ أُقِّتَتۡ ۝١١ لِأَيِّ يَوۡمٍ أُجِّلَتۡ ۝١٢ لِيَوۡمِ ٱلۡفَصۡلِ ۝١٣ وَمَآ أَدۡرَىٰكَ مَا يَوۡمُ ٱلۡفَصۡلِ ۝١٤ وَيۡلٞ يَوۡمَئِذٖ لِّلۡمُكَذِّبِينَ ۝١٥ أَلَمۡ نُهۡلِكِ ٱلۡأَوَّلِينَ ۝١٦ ثُمَّ نُتۡبِعُهُمُ ٱلۡأٓخِرِينَ ۝١٧ كَذَٰلِكَ نَفۡعَلُ بِٱلۡمُجۡرِمِينَ ۝١٨ وَيۡلٞ يَوۡمَئِذٖ لِّلۡمُكَذِّبِينَ ۝١٩ أَلَمۡ نَخۡلُقكُّم مِّن مَّآءٖ مَّهِينٖ ۝٢٠ فَجَعَلۡنَٰهُ فِي قَرَارٖ مَّكِينٍ ۝٢١ إِلَىٰ قَدَرٖ مَّعۡلُومٖ ۝٢٢ فَقَدَرۡنَا فَنِعۡمَ ٱلۡقَٰدِرُونَ ۝٢٣ وَيۡلٞ يَوۡمَئِذٖ لِّلۡمُكَذِّبِينَ ۝٢٤ أَلَمۡ نَجۡعَلِ ٱلۡأَرۡضَ كِفَاتًا ۝٢٥ أَحۡيَآءٗ وَأَمۡوَٰتٗا ۝٢٦ وَجَعَلۡنَا فِيهَا رَوَٰسِيَ شَٰمِخَٰتٖ وَأَسۡقَيۡنَٰكُم مَّآءٗ فُرَاتٗا ۝٢٧ وَيۡلٞ يَوۡمَئِذٖ لِّلۡمُكَذِّبِينَ ۝٢٨ ٱنطَلِقُوٓاْ إِلَىٰ مَا كُنتُم بِهِۦ تُكَذِّبُونَ ۝٢٩ ٱنطَلِقُوٓاْ إِلَىٰ ظِلّٖ ذِي ثَلَٰثِ شُعَبٖ ۝٣٠ لَّا ظَلِيلٖ وَلَا يُغۡنِي مِنَ ٱللَّهَبِ ۝٣١ إِنَّهَا تَرۡمِي بِشَرَرٖ كَٱلۡقَصۡرِ ۝٣٢ كَأَنَّهُۥ جِمَٰلَتٞ صُفۡرٞ ۝٣٣ وَيۡلٞ يَوۡمَئِذٖ لِّلۡمُكَذِّبِينَ ۝٣٤ هَٰذَا يَوۡمُ لَا يَنطِقُونَ ۝٣٥ وَلَا يُؤۡذَنُ لَهُمۡ فَيَعۡتَذِرُونَ ۝٣٦ وَيۡلٞ يَوۡمَئِذٖ لِّلۡمُكَذِّبِينَ ۝٣٧ هَٰذَا يَوۡمُ ٱلۡفَصۡلِۖ جَمَعۡنَٰكُمۡ وَٱلۡأَوَّلِينَ ۝٣٨ فَإِن كَانَ لَكُمۡ كَيۡدٞ فَكِيدُونِ ۝٣٩ وَيۡلٞ يَوۡمَئِذٖ لِّلۡمُكَذِّبِينَ ۝٤٠ إِنَّ ٱلۡمُتَّقِينَ فِي ظِلَٰلٖ وَعُيُونٖ ۝٤١ وَفَوَٰكِهَ مِمَّا يَشۡتَهُونَ ۝٤٢ كُلُواْ وَٱشۡرَبُواْ هَنِيٓ‍َٔۢا بِمَا كُنتُمۡ تَعۡمَلُونَ ۝٤٣ إِنَّا كَذَٰلِكَ نَجۡزِي ٱلۡمُحۡسِنِينَ ۝٤٤ وَيۡلٞ يَوۡمَئِذٖ لِّلۡمُكَذِّبِينَ ۝٤٥ كُلُواْ وَتَمَتَّعُواْ قَلِيلًا إِنَّكُم مُّجۡرِمُونَ ۝٤٦ وَيۡلٞ يَوۡمَئِذٖ لِّلۡمُكَذِّبِينَ ۝٤٧ وَإِذَا قِيلَ لَهُمُ ٱرۡكَعُواْ لَا يَرۡكَعُونَ ۝٤٨ وَيۡلٞ يَوۡمَئِذٖ لِّلۡمُكَذِّبِينَ ۝٤٩ فَبِأَيِّ حَدِيثِۢ بَعۡدَهُۥ يُؤۡمِنُونَ ۝٥٠
